int removeDuplicates(vector<int>& nums)
{
    int fillin = 2;

    for(int i = 2; i < nums.size(); i++)
    {
        if(nums[i] != nums[fillin - 1] || (nums[i] == nums[fillin - 1] && nums[fillin - 1] != nums[fillin - 2]))
        {
            nums[fillin] = nums[i];
            fillin++;
        }
    }
    return nums.size() >= 2 ? fillin : nums.size();
}

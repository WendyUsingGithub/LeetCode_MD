# 73. Set Matrix Zeroes

## Notes
```
因為題目規定需要 in-place
但是不能因此邊 traverse matrix 邊把 matrix 的行或是列改成 0
這會破壞到還沒有 traverse 到的資訊
因此要先記下這些資訊
資訊全部收集完成之後才開始改動 matrix
```

## Code
```
#include <vector>
#include "cout.h"

void setZeroes(vector<vector<int>>& matrix);

int main()
{
    vector<vector<int>> matrix = {{0, 1, 2, 0}, {3, 4, 5, 2}, {1, 3, 1, 5}};
    
    setZeroes(matrix);
    coutVectorVector(matrix);
    return 0;
}

void setZeroes(vector<vector<int>>& matrix)
{
    int m = matrix.size();
    int n = matrix[0].size();
    vector<int> rows(m, 0);  
    vector<int> cols(n, 0); 

    for(int i = 0; i < m; i++)
    {
        for(int j = 0; j < n; j++)
        {
            if(matrix[i][j] == 0)
            {
                rows[i] = 1;
                cols[j] = 1;
            }
        }
    } 

    for(int i = 0; i < m; i++)
    {
        if(rows[i] == 1)
        {
            for(int j = 0; j < n; j++)
            {
                matrix[i][j] = 0;
            }
        }
    }

    for(int i = 0; i < n; i++)
    {
        if(cols[i] == 1)
        {
            for(int j = 0; j < m; j++)
            {
                matrix[j][i] = 0;
            }
        }
    }
    return;
}
```
